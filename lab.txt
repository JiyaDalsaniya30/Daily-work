import numpy as np
#indexing and slicing in 1D array
a=np.array([10,20,30,40,50])
#indexing
print("a[1] :", a[1])
print("a[2] :", a[2]) #getting only element
print("a[-1] :", a[-1])
print("\n")

#slicing
#araay[start : stop : step]
print("a[1:3] : ",a[1 : 3])
print("a[-3:-1] : ",a[-3 : -1]) #getting array
print("a[1:2] : ", a[1 : 2])
print("a[0:4:2] : ", a[0 : 4 : 2])
print("a[0:4:2] : ", a[0 : 4 : -2])
print("a[::2] : ", a[ :  : -1]) #for reverse the array
